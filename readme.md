#Application.properties file
add database properties in application.property

server.port=<port no>
driverName=com.mysql.jdbc.Driver
dbUrl=<database url>
username=user name
password=password
hbmToDdl=none
hibernateDilact=org.hibernate.dialect.MySQL5Dialect
